# Sets up SDR in loopback mode to transmit a preloaded csv signal to RX and write rcvd signal to csv.
# This list of commands must be entered in this order.
# Will be used in the terminal with bladeRF
set loopback firmware
set frequency tx 2400M
set frequency rx 2400M
set samplerate tx 30M
set samplerate rx 30M
tx config channel=1 file=PATH/sine_sig.csv format=csv repeat=6 timeout=30s
rx config file=PATH/rcvd_sig.csv format=csv n=6000 timeout=30s
rx start
tx start
